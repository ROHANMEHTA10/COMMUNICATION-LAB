function send() { let input = document.getElementById("input"); let text = input.value.toLowerCase(); if(text === "") return; let chat = document.getElementById("chat"); chat.innerHTML += `
${text}
`; input.value = ""; let reply = ""; if(text.includes("hi") || text.includes("hello")) { reply = "Hey! 😊 Tell me your introduction."; } else if(text.includes("name")) { reply = "Nice! Now tell me what you do."; } else if(text.includes("bad") || text.includes("abuse")) { reply = "Hey 😅 let's keep it respectful. Focus on learning."; } else { reply = "Good try 👍 Speak clearly and confidently. Try again."; } setTimeout(() => { chat.innerHTML += `
${reply}
`; }, 1000); }
