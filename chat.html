function send() { let input = document.getElementById("input"); let text = input.value.toLowerCase(); if(text === "") return; let chat = document.getElementById("chat"); // user message chat.innerHTML += `
${text}
`; input.value = ""; let reply = ""; if(text.includes("hi") || text.includes("hello")) { reply = "Hey 😊 Tell me your name."; } else if(text.includes("name")) { reply = "Nice! What do you do?"; } else if(text.includes("student") || text.includes("work")) { reply = "Great 👍 What are your interests?"; } else if(text.includes("interest")) { reply = "Awesome 🔥 You're doing great. Try speaking confidently!"; } else if(text.includes("bad") || text.includes("abuse")) { reply = "Hey 😅 let's keep it respectful."; } else { reply = "Good 👍 Try giving full introduction: Name, work, interest."; } setTimeout(() => { chat.innerHTML += `
${reply}
`; chat.scrollTop = chat.scrollHeight; }, 800); }
